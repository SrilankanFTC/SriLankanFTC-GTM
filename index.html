function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  var s = sheet.getSheetByName(data.type) || sheet.insertSheet(data.type);
  s.clearContents();
  data.headers && s.appendRow(data.headers);
  data.rows.forEach(function(r){ s.appendRow(r); });
  return ContentService.createTextOutput('OK');
}
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var out = {};
  ['sessions','instructors','classes'].forEach(function(n){
    var s = sheet.getSheetByName(n);
    if(s) out[n] = s.getDataRange().getValues();
  });
  return ContentService.createTextOutput(JSON.stringify(out))
    .setMimeType(ContentService.MimeType.JSON);
}
